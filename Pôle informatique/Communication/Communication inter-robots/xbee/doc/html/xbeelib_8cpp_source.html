351int XBee::crc16(int trame[], uint8_t taille){
352 int crc = 0xFFFF, count = 0;
353 int octet_a_traiter;
354 const int POLYNOME = 0xA001;
355
356 octet_a_traiter = trame[0];
357
358 do{
359 crc ^= octet_a_traiter;
360 for(uint8_t i = 0; i < 8; i++){
361
362 if((crc%2)!=0)
363 crc = (crc >> 1) ^ POLYNOME;
364
365 else
366 crc = (crc >> 1);
367
368 }
369 count++;
370 octet_a_traiter = trame[count];
371
372 }while(count < taille);
373
374 return crc;
375}

437int XBee::processTrame(vector<int> trame_recue){
438
439 if(!isTrameSizeCorrect(trame_recue)){
440 logXbee << "/!\\ (process trame) erreur " << XB_TRAME_E_SIZE << " : taille de la trame incorrecte ou non concordante " << mendl;
441 return XB_TRAME_E_SIZE;
442 }
443 Trame_t trame = {
444 .start_seq = trame_recue[0],
445 .adr_emetteur = trame_recue[1],
446 .adr_dest = trame_recue[2],
447 .id_trame_low = trame_recue[3]-4,
448 .id_trame_high = trame_recue[4]-4,
449 .nb_octets_msg = trame_recue[5]-4,
450 .code_fct = trame_recue[6],
451 .crc_low = trame_recue[3+trame_recue[4]],
452 .crc_high = trame_recue[4+trame_recue[4]],
453 .end_seq = trame_recue[5+trame_recue[4]]
454 };
455
456 vector<int> data {};
457
458 for(uint8_t i = 0; i < trame.nb_octets_msg; i++){
459 data.push_back(trame_recue[7+i]);
460 }
461
462 trame.param = data;
463
464 afficherTrameRecue(trame);
465
466 int decoupe_trame[trame_recue[4]+6];
467
468 for(uint8_t i = 0; i < trame_recue[4]+3; i++){
469 decoupe_trame[i] = trame_recue[i];
470 }
471
472 if(!isStartSeqCorrect(trame.start_seq)){
473 logXbee << "/!\\ (process trame) erreur " << XB_TRAME_E_START << " : premier caractère de la trame incorrect " << mendl;

478 logXbee << "/!\\ (process trame) erreur " << XB_TRAME_E_END << " : dernier caractère de la trame incorrect " << mendl;
479 return XB_TRAME_E_END;

483 logXbee << "/!\\ (process trame) erreur " << XB_TRAME_E_CRC << " : valeur du CRC incorrecte " << mendl;

485 }
486
487 if(!isExpCorrect(trame.adr_emetteur)){
488 logXbee << "/!\\ (process trame) erreur " << XB_TRAME_E_EXP << " : adresse de l'expéditeur incorrecte ou inconnue " << mendl;
489 return XB_TRAME_E_EXP;
490 }
491

493 logXbee << "/!\\ (process trame) erreur " << XB_TRAME_E_DEST << " : adresse du destinataire incorrecte ou inconnue " << mendl;
494 return XB_TRAME_E_DEST;
495 }
496
497 processCodeFct(trame.code_fct, trame.adr_emetteur);
498
499 logXbee << "(process trame) trame n°" << trame.id_trame_high+trame.id_trame_low << "a été traitée avec succès " << mendl;
500
501 return XB_TRAME_E_SUCCESS;

525 }
526
527 trames_envoyees[code_fct] = trames_envoyees[code_fct]-1;
528 logXbee << "(process code fonction) code fonction n°" << code_fct << " traité avec succès" << mendl;
529 return XB_FCT_E_SUCCESS;

725int XBee::subTrame(vector<int> msg_recu){
726
727 vector<int> list_start_seq {};
728 vector<int> list_end_seq {};
729 vector<int> decoupe {};
730 int decoupe_retour;
731
732 for(uint8_t i = 0; i < msg_recu.size(); i++){
733 if(msg_recu[i] == XB_V_START)
734 list_start_seq.push_back(i);
735
736 if(msg_recu[i] == XB_V_END)
737 list_end_seq.push_back(i);
738 }
739
740 if(list_start_seq.size() == 0 || list_end_seq.size() == 0){
741 logXbee << "/!\\ (découpe trame) erreur " << XB_SUB_TRAME_E_NULL << " : aucun caractère de début et/ou de fin n'est présent dans le message reçu " << mendl;
742 return XB_SUB_TRAME_E_NULL;
743 }
744
745 if(list_start_seq.size() != list_end_seq.size()){
746 logXbee << "/!\\ (découpe trame) erreur " << XB_SUB_TRAME_E_SIZE << " : les caractères de début et de fin de trame ne sont pas au même nombre " << mendl;
747 return XB_SUB_TRAME_E_SIZE;
748 }
749
750 for(uint8_t i = 0; i < list_start_seq.size(); i++){
751 if(list_start_seq[i] > list_end_seq[i]){
752 logXbee << "/!\\ (découpe trame) erreur " << XB_SUB_TRAME_E_REPARTITION << " : certains caractères de début de trame sont placés après des caractères de fin de trame " << mendl;
